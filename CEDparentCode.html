SELECT M.Project_ID,
       LEFT(Project_Name, 99) AS Project_Name,
       P.FiscalYearCompleted,
       P.Status AS Effort_Status,
       P.ProjectStartDate AS Start_Date,
       P.ProjectEndDate AS End_Date,
       'Acres' AS Metric_Type,
       NULL AS Metric_Value,
       NULL AS GIS_Acres,
       CASE
           WHEN PC.CategoryOther = 'Fire Rehab' THEN 'Yes'
           ELSE 'No'
       END AS Post_Fire,
       '9' AS Implementing_Party,
       '116' AS Office,
       'Tyler Thompson' AS Created_By,
       '0' AS NumberDocs,
       PCF.CategoryFundingID,
       'Yes' AS Mitigation_Status,
       CASE
           WHEN ISNULL(U.state, 0) = 0 THEN ISNULL(U.notstate, 0) + 1
           ELSE ISNULL(U.notstate, 0)
       END AS NumberCollaborators,
       '3' AS Entry_Status,
       '3' AS Effect_Status,
       ISNULL(NumberThreatsAddress, 0) AS NumberThreatsAddress,
       num AS NumberLandOwners,
       CASE
           WHEN PCG.CategoryOther = 'Sagebrush Restoration' THEN 'Yes'
           ELSE 'No'
       END AS SagebrushRestoration,
       Seeding_Type,
       Subactivity,
       Activity,
       M.FeatureID,
       Shape
FROM
  (SELECT Project_Name,
          Project_ID,
          Seeding_Type,
          Subactivity,
          FeatureID,
          CASE
              WHEN M.Subactivity = 17 THEN 9
              WHEN M.Subactivity = 34 THEN 11
              WHEN M.Subactivity = 35 THEN 11
              WHEN M.Subactivity = 36 THEN 11
              ELSE 10
          END AS Activity
   FROM
     (SELECT M.Project_ID,
             Seeding_Type,
             Project_Name,
             CASE
                 WHEN M.ActionID = 2
                      AND M.CategoryOtherID = 1 THEN 33
                 WHEN M.ActionID = 4
                      AND M.CategoryOtherID = 3 THEN 32
                 WHEN M.ActionID = 5
                      AND M.CategoryOtherID = 3 THEN 32
                 WHEN M.ActionID = 7 THEN 33
                 WHEN M.ActionID = 9
                      AND M.CategoryOtherID > 8 THEN 30
                 WHEN M.ActionID = 9
                      AND M.CategoryOtherID < 9 THEN 33
                 WHEN M.ActionID = 12
                      AND M.CategoryOtherID = 3 THEN 32
                 WHEN M.ActionID = 13 THEN 33
                 WHEN M.ActionID = 15 THEN 33
                 WHEN M.ActionID = 16 THEN 30
                 WHEN M.ActionID = 17 THEN 33
                 WHEN M.ActionID = 18 THEN 33
                 WHEN M.ActionID = 22
                      AND M.CategoryOtherID > 8 THEN 30
                 WHEN M.ActionID = 22
                      AND M.CategoryOtherID < 9 THEN 33
                 WHEN M.ActionID = 24 THEN 33
                 WHEN M.ActionID = 25
                      AND M.CategoryOtherID = 3 THEN 32
                 WHEN M.ActionID = 25
                      AND M.CategoryOtherID > 8 THEN 30
                 WHEN M.ActionID = 25
                      AND M.CategoryOtherID < 3 THEN 33
                 WHEN M.ActionID = 9
                      AND M.CategoryOtherID > 2
                      AND M.CategoryOtherID < 9 THEN 33
                 WHEN M.ActionID = 28 THEN 33
                 WHEN M.ActionID = 32 THEN 33
                 WHEN M.ActionID = 33 THEN 33
                 WHEN M.ActionID = 34 THEN 33
                 WHEN M.ActionID = 37
                      AND M.CategoryOtherID > 8 THEN 30
                 WHEN M.ActionID = 37
                      AND M.CategoryOtherID < 9 THEN 33
                 WHEN M.ActionID = 38
                      AND M.CategoryOtherID > 8 THEN 30
                 WHEN M.ActionID = 38
                      AND M.CategoryOtherID < 9
                      AND M.CategoryOtherID > 2 THEN 33
                 WHEN M.ActionID = 38
                      AND M.CategoryOtherID = 1 THEN 33
                 ELSE 17
             END AS Subactivity,
             M.FeatureID
      FROM
        (SELECT P.Project_ID,
                CASE
                    WHEN A.ActionDescription IN ('Seeding (primary)',
                                                 'Seeding (secondary/shrub)') THEN 'Native/Non-Native Mixed'
                END AS Seeding_Type,
                CONCAT(P.Title, '_', A.ActionDescription) AS Project_Name,
                SA.CategoryOtherID,
                A.ActionID AS ActionID,
                A.FeatureID AS FeatureID
         FROM WRI.dbo.AREAACTION AS A
         LEFT JOIN WRI.dbo.POLY AS PO ON PO.FeatureID = A.FeatureID
         LEFT JOIN WRI.dbo.PROJECT AS P ON PO.Project_ID = P.Project_ID
         LEFT JOIN WRI.dbo.PROJECTCATEGORYOTHER AS SA ON P.Project_ID = SA.Project_ID) AS M
      LEFT JOIN WRI.dbo.Methods_Domain AS MD ON MD.ActionID = M.ActionID) AS M) AS M
LEFT JOIN
  (SELECT Project_Named,
          Count(ADMIN) AS num
   FROM
     (SELECT DISTINCT CONCAT(P.Title, '_', A.ActionDescription) AS Project_Named,
                      ADMIN
      FROM WRI.dbo.AREAACTION AS A
      LEFT JOIN WRI.dbo.POLY AS PO ON PO.FeatureID = A.FeatureID
      LEFT JOIN
        (SELECT FeatureID,
                CASE
                    WHEN OWNER = 'Federal' THEN ADMIN
                    ELSE OWNER
                END AS ADMIN
         FROM WRI.dbo.LANDOWNER) AS L ON PO.FeatureID = L.FeatureID
      LEFT JOIN WRI.dbo.PROJECT AS P ON PO.Project_ID = P.Project_ID) AS LO
   GROUP BY Project_Named) AS L ON L.Project_Named = M.Project_Name
LEFT JOIN WRI.dbo.PROJECT AS P ON M.Project_ID = P.Project_ID
LEFT JOIN WRI.dbo.POLY AS PO ON PO.FeatureID = M.FeatureID
LEFT JOIN
  (SELECT Project_FK,
          Count(DISTINCT (CASE
                              WHEN Agency = 9 THEN Agency
                          END)) AS state,
          Count(DISTINCT Agency) AS notstate
   FROM
     (SELECT Project_FK,
             User_FK,
             CED_Code AS Agency
      FROM WRI.dbo.CONTRIBUTOR
      RIGHT JOIN
        (SELECT CED_Code,
                User_ID
         FROM WRI.dbo.USERS AS X
         LEFT JOIN WRI.dbo.Agency_Domain AS Y ON X.Agency = Y.WRI_AgencyID) AS DOMAIN ON User_ID = User_FK) AS users
   GROUP BY Project_FK) AS U ON U.Project_FK = P.Project_ID
LEFT JOIN
  (SELECT *
   FROM WRI.dbo.PROJECTCATEGORYOTHER
   WHERE CategoryOther = 'Fire Rehab') AS PC ON PC.Project_ID = P.Project_ID
LEFT JOIN
  (SELECT *
   FROM WRI.dbo.PROJECTCATEGORYOTHER
   WHERE CategoryOther = 'Sagebrush Restoration') AS PCG ON PCG.Project_ID = P.Project_ID
LEFT JOIN
  (SELECT Project_ID,
          COUNT(DISTINCT CED_Code) AS NumberThreatsAddress
   FROM WRI.dbo.SPECIES
   LEFT JOIN
     (SELECT TD.Threat_ID,
             CED_Code,
             Species_ID,
             Impact_ID
      FROM WRI.dbo.SPECIES_THREAT_IMPACT AS TI
      LEFT JOIN WRI.dbo.Threats_Domain AS TD ON TD.Threat_ID = TI.Threat_ID
      WHERE CED_Code > 0) AS TITD ON SPECIES.Threat_ID = TITD.Threat_ID
   WHERE Project_ID IS NOT NULL
     AND CED_Code IS NOT NULL
   GROUP BY Project_ID) AS T ON T.Project_ID = P.Project_ID
LEFT JOIN WRI.dbo.PROJECTCATEGORYFUNDING AS PCF ON PCF.Project_ID = P.Project_ID
WHERE PCF.CategoryFundingID = 1
